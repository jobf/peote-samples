package;

import haxe.CallStack;

import lime.app.Application;
import lime.ui.Window;

import peote.view.PeoteView;
import peote.view.Buffer;
import peote.view.Display;
import peote.view.Program;
import peote.view.Color;


class Main extends Application
{
	override function onWindowCreate():Void
	{
		switch (window.context.type)
		{
			case WEBGL, OPENGL, OPENGLES:
				try startSample(window)
				catch (_) trace(CallStack.toString(CallStack.exceptionStack()), _);
			default: throw("Sorry, only works with OpenGL.");
		}
	}
	
	// ------------------------------------------------------------
	// --------------- SAMPLE STARTS HERE -------------------------
	// ------------------------------------------------------------
	var buffer:Buffer;
	var lineSegment:LineSegment;
	
	var isInit = false;
	
	public function startSample(window:Window)
	{
		var peoteView = new PeoteView(window);

		buffer = new Buffer(4, 4, true);
		var display = new Display(0, 0, window.width, window.height, Color.GREEN);
		var program = new Program(buffer);

		peoteView.addDisplay(display);
		display.addProgram(program);

		
		lineSegment = new LineSegment();
		
		// line starts at middle of the window
		lineSegment.x = Std.int(window.width / 2);
		lineSegment.y = Std.int(window.height / 2);
		lineSegment.h = 3;
		
		// pivot point for rotation and positionizing
		lineSegment.px = 0;
		lineSegment.py = 1;
		
		buffer.addElement(lineSegment);
		
		isInit = true;
	}
	
	// ----------------- MOUSE EVENTS ------------------------------
	override function onMouseMove (x:Float, y:Float):Void {
		if (isInit) 
		{
			var a = lineSegment.x - x;
			var b = lineSegment.y - y;
			
			lineSegment.w = Std.int( Math.sqrt( a * a + b * b ) );
			lineSegment.r = Math.atan2(x - lineSegment.x, - (y - lineSegment.y) )*(180 / Math.PI) - 90; // thx to halfwheat \o/
			
			buffer.updateElement(lineSegment);
		}
	}
	
	// -------------- WINDOWS EVENTS ----------------------------
	override function onWindowResize (width:Int, height:Int):Void {
		if (isInit) 
		{	// keep line starting at middle of the window on resize
			lineSegment.x = Std.int(width / 2);
			lineSegment.y = Std.int(height / 2);
		}
	}
	
	// override function onMouseDown (x:Float, y:Float, button:lime.ui.MouseButton):Void {}	
	// override function onMouseUp (x:Float, y:Float, button:lime.ui.MouseButton):Void {}

	// override function onMouseMoveRelative (x:Float, y:Float):Void {}
	
	// ----------------- KEYBOARD EVENTS ---------------------------

}
